git :

#将文件添加到暂缓区中
git add

#将文件添加到代码库中
git commit -m "xxx"

#查看日志
git log
git log --pretty=oneline

#查看状态
git status

git diff 
#查看工作区和版本库中的区别
git diff HEAD -- readme.txt  

#版本回退
git reset --hard HEAD^
git reset --hard commit_id

git reflog

git rm 

#在本地操作，将本地仓库与远程仓库关联
git remote add origin git@github.com:michaelliao/learngit.git
#在远程创建了远程仓库后，与本地仓库相关联
git clone git@github.com:michaelliao/gitskills.git

#将本地仓库代码推送到远程仓库的主分支
git push -u origin master

#将远程库中的对应分支代码下载到本地对应分支
git clone -b 分支名 远程库地址

#创建dev分支，并且切换到dev分支
git checkout -b <name>
example:git checkout -b dev

#创建分支
git branch <name>
example:git branch dev

#切换分支
git checkout <name>
example:git checkout dev

#查看分支
git branch

#合并指定分支到当前分支。
#所以先切换到master分之后，再使用git merge
git merge <name>
example:git merge dev

#删除分支
git branch -d <name>
example:git branch -d dev

#使用非fast-forward模式进行合并
git merge --no-ff -m "merge with no-ff" dev
